import turtle
#square
qazi_turtle=turtle.Turtlet()

def square():
   qazi_turtle.forward(100)
   qazi_turtle.right(90)
   qazi_turtle.forward(100)
   qazi_turtle.right(90)
   qazi_turtle.forward(100)
   qazi_turtle.right(90)
   qazi_turtle.forward(100)

square()
qazi_turtle.forward(200)
square()
